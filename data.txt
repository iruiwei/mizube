
注意：
データベースを選んだあとからsqlを実行してください。
テーブルを１個つづ作ってください。
３つのデータは同時にインサートできます。

create table mb_restaurant (
	rid int,
	tag_id text,
	lat double,
	lon double,
	name varchar(50),
	view int,
	uniqueuser int,
	area_id int,
	menu text,
	opentime datetime,
	closetime datetime,
	photo varchar(100),
	phone varchar(30),
	introduction text,
	comment text
)


create table mb_area(
	aid int,
	area_name varchar(50),
	lat double,
	lon double,
	hashtag int
)


create table mb_shiptime(
	sid int,
	departureID int,
	arrivalID int,
	depTime datetime,
	costTime datetime,
	maxTicket1 int,
	currentTicket1 int,
	maxTicket2 int,
	currentTicket2 int,
	train	text,
	walk	text,
	comment	text
)


create table mb_tag(
	tag_id int,
	tag_name varchar(50)
)
